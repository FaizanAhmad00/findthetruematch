CREATE TABLE Candidate(CANDIDATE_ID int AUTO_INCREMENT PRIMARY KEY, EMAIL varchar(50) DEFAULT NULL, 
	PASSWORD varchar(50) DEFAULT NULL, MEMBERSHIP int DEFAULT NULL, NAME varchar(50) DEFAULT NULL, AGE int DEFAULT NULL, 
	HEIGHT decimal(5,2) DEFAULT NULL, CAST varchar(50) DEFAULT NULL, RELIGION varchar(50) DEFAULT NULL, FATHER_NAME varchar(50) DEFAULT NULL, 
	MOTHER_NAME varchar(50) DEFAULT NULL, CONTACT_NUMBER varchar(50) DEFAULT NULL, CITY varchar(50) DEFAULT NULL, MARITAL_STATUS varchar(50) DEFAULT NULL, 
	GENDER varchar(50) DEFAULT NULL, PROFILE_IMAGE blob DEFAULT NULL);

CREATE TABLE family_details (SIBLINGS int DEFAULT 0, FATHER_OCCUPATION varchar(50) DEFAULT NULL, MOTHER_OCCUPATION varchar(50) DEFAULT NULL, 
	MARRIED_FAMILY_MEMBERS int DEFAULT NULL, CANDIDATE_ID_FK int DEFAULT 0);

CREATE TABLE property_details (NO_OF_HOUSES int DEFAULT 0, NO_OF_SHOPS int DEFAULT NULL, LAND_PLOT varchar(50) DEFAULT NULL, CARS_AND_VEHICLES varchar(50) DEFAULT NULL,
	FACTORY_AND_VEHICLES varchar(50) DEFAULT NULL, CANDIDATE_ID_FK int DEFAULT 0);

CREATE TABLE job_details (COMPANY_NAME varchar(50), OCCUPATION varchar(50) DEFAULT NULL, INCOME int DEFAULT NULL, DESIGNATION varchar(50) DEFAULT NULL, 
	CANDIDATE_ID_FK int DEFAULT 0);

CREATE TABLE education_details (COMPLETION_YEAR int DEFAULT 0, QUALIFICATION varchar(50) DEFAULT NULL, LAST_INSTITUTE varchar(50) DEFAULT NULL, 
	OTHER_ACHIVEMENT varchar(50) DEFAULT NULL, CANDIDATE_ID_FK int DEFAULT 0);

CREATE TABLE address_city(ADDRESS_ID int PRIMARY KEY, ADDRESS varchar(50) DEFAULT null, CITY varchar(50));

CREATE TABLE sect_religion(SECT_ID int PRIMARY KEY, SECT varchar(50) DEFAULT null, RELIGION varchar(50) DEFAULT NULL);
